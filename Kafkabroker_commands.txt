1. Configure Kafka server
cd/opt/kafka
nano config/server.properties


Update:
# Allow connections from other machines
listeners=PLAINTEXT://0.0.0.0:9092
# Advertise your machine's IP (replace with actual IP)
advertised.listeners=PLAINTEXT://172.27.173.202:9092


2. Start zookeeper and kafka:
cd /opt/kafka
bin/zookeeper-server-start.sh config/zookeeper.properties


On a new terminal:
cd /opt/kafka
bin/kafka-server-start.sh config/server.properties


3. Kafka topic creation and management
 Create Topics
               cd /opt/kafka            


     bin/kafka-topics.sh --create \
    --bootstrap-server localhost:9092 \
    --topic tasks \
    --partitions 2 \
    --replication-factor 1 \
    --config retention.ms=3600000


bin/kafka-topics.sh --create \
    --bootstrap-server localhost:9092 \
    --topic results \
    --partitions 2 \
    --replication-factor 1 \
    --config retention.ms=3600000


bin/kafka-topics.sh --create \
    --bootstrap-server localhost:9092 \
    --topic heartbeats \
    --partitions 1 \
    --replication-factor 1 \
    --config retention.ms=300000




 Delete Topics
cd /opt/kafka


bin/kafka-topics.sh --delete --bootstrap-server localhost:9092 --topic tasks
bin/kafka-topics.sh --delete --bootstrap-server localhost:9092 --topic results
bin/kafka-topics.sh --delete --bootstrap-server localhost:9092 --topic heartbeats




 List Topics
cd /opt/kafka


bin/kafka-topics.sh --list --bootstrap-server localhost:9092
